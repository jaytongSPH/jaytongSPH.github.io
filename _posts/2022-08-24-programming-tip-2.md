git clone https://github.com/jaytongSPH/jaytongSPH.github.io

git config --list

git pull origin main 下载代码并合并

git add --all

git commit -m "update"

git push origin main  上传代码并合并

git branch 查看本地项目的分支

git branch -r 查看远程项目的分支

git branch -d 删除本地项目的分支

git checkout 显示分支名称
